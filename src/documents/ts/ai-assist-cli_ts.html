CLI-based AI assistant using Typescript

File: scripts/cli_assistant.ts

import OpenAI from "openai";
import readline from "readline";

// Bun natively injects parameters from your local .env file
const apiKey = Bun.env.OPENAI_API_KEY;

if (!apiKey) {
  console.error("Error: Missing OPENAI_API_KEY. Put it in a .env file.");
  process.exit(1);
}

const client = new OpenAI({ apiKey });

/**
 * Builds a structured message payload for the LLM.
 * Separates system instructions from user input.
 */
function buildMessages(userInput: string) {
  return [
    {
      role: "system" as const,
      content: "You are a helpful AI assistant. Be concise, accurate, and clear. If you are unsure, say you are unsure."
    },
    {
      role: "user" as const,
      content: userInput
    }
  ];
}

/**
 * Sends a request to the LLM API and returns the text output.
 */
async function callLlm(userInput: string): Promise<string> {
  const messages = buildMessages(userInput);

  const response = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: messages,
    temperature: 0.2,
    max_tokens: 300,
  });

  return response.choices[0].message.content?.trim() || "";
}

async function main() {
  console.log("CLI AI Assistant");
  console.log("Type 'exit' or 'quit' to end the session.\n");

  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
  });

  const askQuestion = () => {
    rl.question("You: ", async (input) => {
      const userInput = input.trim();

      if (userInput.toLowerCase() === "exit" || userInput.toLowerCase() === "quit") {
        console.log("Goodbye.");
        rl.close();
        process.exit(0);
      }

      if (!userInput) {
        console.log("Please enter a prompt.");
        askQuestion();
        return;
      }

      try {
        const answer = await callLlm(userInput);
        console.log(`\nAssistant: ${answer}\n`);
      } catch (e: any) {
        console.error("\nError calling the AI model:");
        console.error(e.message || e, "\n");
      }

      askQuestion();
    });
  };

  askQuestion();
}

main();

How to Run

Initialize project packages and launch the script immediately from the project root using Bun:

bun install
bun run scripts/cli_assistant.ts
